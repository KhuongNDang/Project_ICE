DROP TABLE IF EXISTS Potion;
DROP TABLE IF EXISTS Creature;
DROP TABLE IF EXISTS Item;
DROP TABLE IF EXISTS AsciiCreature;
DROP TABLE IF EXISTS AsciiGear;
DROP TABLE IF EXISTS AsciiLocation;
DROP TABLE IF EXISTS Location;


CREATE TABLE IF NOT EXISTS Potion
(
    id      INTEGER PRIMARY KEY AUTOINCREMENT,
    name    TEXT NOT NULL,
    attack  INTEGER DEFAULT 0,
    defense INTEGER DEFAULT 0,
    health  INTEGER DEFAULT 0
);


CREATE TABLE Creature
(
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        varchar(20) NOT NULL,
    attack      INTEGER     NOT NULL,
    defense     INTEGER     NOT NULL,
    health      INTEGER     NOT NULL,
    xp          INTEGER     NOT NULL,
    currency    TEXT        NOT NULL
);


CREATE TABLE Item
 (  id      INTEGER PRIMARY KEY AUTOINCREMENT,
    name    TEXT    NOT NULL,
    attack  INTEGER DEFAULT 0,
    defense INTEGER DEFAULT 0,
    health  INTEGER DEFAULT 0,
    slot    VARCHAR(20)    NOT NULL
);


CREATE TABLE AsciiCreature
(
    id      INTEGER PRIMARY KEY AUTOINCREMENT,
    picture TEXT NOT NULL
);

CREATE TABLE AsciiGear
(
    id      INTEGER PRIMARY KEY AUTOINCREMENT,
    picture TEXT NOT NULL
);

CREATE TABLE AsciiLocation
(
    id      INTEGER PRIMARY KEY AUTOINCREMENT,
    picture TEXT NOT NULL
);

CREATE TABLE Location
(  id      INTEGER PRIMARY KEY AUTOINCREMENT,
   name    TEXT    NOT NULL
);
